    (let ((cur (current-buffer))
          (tfmt "%s")
          (i 0))
      (with-current-buffer cur
        (goto-char (point-max))
        (while (< i (* 10 86400))
          (if (not (eq i 0)) (insert "\n"))
          (insert (format "filename%d.jpg," (- 10 (/ i 86400))))
          (insert (number-to-string (- (string-to-number (format-time-string tfmt (current-time))) i)))
          (setq i (+ i 86400)))))
